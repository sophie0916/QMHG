  1. Haldane Mapping Distance Calculation
getHaldane <- function(rf) {
  rfValue <- rf/100
  m <- -log(1-2*rfValue)
  mdHaldane <- 50*m
  return(mdHaldane)
}

hDist1_12 <- getHaldane(mDist1_12)
